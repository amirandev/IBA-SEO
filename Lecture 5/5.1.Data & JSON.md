```jsx
// src/components/ProductCard.jsx
import React from 'react';
import { Link } from 'react-router-dom';
import { useLanguage } from '../contexts/LanguageContext';

const ProductCard = ({ product }) => {
  const { lang } = useLanguage();
  
  const discountedPrice = product.price.sale || product.price.regular;
  const hasDiscount = product.price.sale !== null;
  const discountPercent = hasDiscount 
    ? Math.round((1 - product.price.sale / product.price.regular) * 100)
    : 0;

  return (
    <div className={`product-card ${!product.inStock ? 'out-of-stock' : ''}`}>
      {/* ბეჯები / Badges */}
      <div className="badges">
        {product.newArrival && (
          <span className="badge new">
            {lang === 'ka' ? 'ახალი' : 'New'}
          </span>
        )}
        {hasDiscount && (
          <span className="badge sale">-{discountPercent}%</span>
        )}
      </div>

      {/* სურათი / Image */}
      <Link to={`/products/${product.slug}`} className="image-link">
        <img
          src={product.images.find(img => img.isPrimary)?.url || product.images[0]?.url}
          alt={product.name[lang]}
          loading="lazy"
          className="product-image"
        />
      </Link>

      {/* ინფორმაცია / Info */}
      <div className="product-info">
        <Link to={`/products/${product.slug}`} className="product-link">
          <h3 className="product-name">{product.name[lang]}</h3>
        </Link>

        <p className="product-description">
          {product.shortDescription[lang]}
        </p>

        {/* რეიტინგი / Rating */}
        <div className="rating">
          <span className="stars">
            {'⭐'.repeat(Math.round(product.rating))}
          </span>
          <span className="rating-text">
            {product.rating} ({product.reviewCount})
          </span>
        </div>

        {/* ფასი / Price */}
        <div className="price-container">
          {hasDiscount && (
            <span className="old-price">
              {product.price.regular} ₾
            </span>
          )}
          <span className="current-price">
            {discountedPrice} ₾
          </span>
        </div>

        {/* ღილაკი / Button */}
        <button
          className={`add-to-cart-btn ${!product.inStock ? 'disabled' : ''}`}
          disabled={!product.inStock}
        >
          {product.inStock
            ? (lang === 'ka' ? ' კალათაში დამატება' : '🛒 Add to Cart')
            : (lang === 'ka' ? 'მარაგში არ არის' : 'Out of Stock')
          }
        </button>
      </div>
    </div>
  );
};

export default ProductCard;
```

## 🎯 პროდუქტების სიის გვერდი / PRODUCTS LIST PAGE

```jsx
// src/pages/ProductsPage.jsx
import React, { useState, useMemo } from 'react';
import { Helmet } from 'react-helmet-async';
import SEO from '../components/SEO';
import ProductCard from '../components/ProductCard';
import { useProducts } from '../hooks/useProducts';
import { useLanguage } from '../contexts/LanguageContext';

const ProductsPage = () => {
  const { lang } = useLanguage();
  const { products, categories, loading, error } = useProducts();
  
  const [selectedCategory, setSelectedCategory] = useState('all');
  const [sortBy, setSortBy] = useState('newest');
  const [searchQuery, setSearchQuery] = useState('');

  // ფილტრაცია და სორტირება / Filtering and sorting
  const filteredProducts = useMemo(() => {
    let result = [...products];

    // კატეგორიის ფილტრი / Category filter
    if (selectedCategory !== 'all') {
      result = result.filter(p => p.category === selectedCategory);
    }

    // ძიება / Search
    if (searchQuery.trim()) {
      const query = searchQuery.toLowerCase();
      result = result.filter(p =>
        p.name[lang].toLowerCase().includes(query) ||
        p.description[lang].toLowerCase().includes(query) ||
        p.tags.some(tag => tag.toLowerCase().includes(query))
      );
    }

    // სორტირება / Sorting
    switch (sortBy) {
      case 'price-low':
        result.sort((a, b) => (a.price.sale || a.price.regular) - (b.price.sale || b.price.regular));
        break;
      case 'price-high':
        result.sort((a, b) => (b.price.sale || b.price.regular) - (a.price.sale || a.price.regular));
        break;
      case 'rating':
        result.sort((a, b) => b.rating - a.rating);
        break;
      case 'newest':
      default:
        result.sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt));
    }

    return result;
  }, [products, selectedCategory, sortBy, searchQuery, lang]);

  if (loading) {
    return (
      <div className="loading-container">
        <div className="spinner"></div>
        <p>{lang === 'ka' ? 'იტვირთება...' : 'Loading...'}</p>
      </div>
    );
  }

  if (error) {
    return (
      <div className="error-container">
        <p> {error}</p>
        <button onClick={() => window.location.reload()}>
          {lang === 'ka' ? 'თავიდან ცდა' : 'Try Again'}
        </button>
      </div>
    );
  }

  return (
    <>
      <SEO
        title={lang === 'ka' ? 'პროდუქტები' : 'Products'}
        description={lang === 'ka' 
          ? 'შეიძინეთ ელექტრონიკა საუკეთესო ფასებით' 
          : 'Buy electronics at best prices'}
        url="https://yourdomain.ge/products"
      />

      <div className="products-page">
        <h1 className="page-title">
          {lang === 'ka' ? 'ჩვენი პროდუქტები' : 'Our Products'}
        </h1>

        {/* ფილტრები / Filters */}
        <div className="filters">
          {/* კატეგორიები / Categories */}
          <select
            value={selectedCategory}
            onChange={(e) => setSelectedCategory(e.target.value)}
            className="filter-select"
          >
            <option value="all">
              {lang === 'ka' ? 'ყველა კატეგორია' : 'All Categories'}
            </option>
            {categories.map(cat => (
              <option key={cat.id} value={cat.slug}>
                {cat.name[lang]}
              </option>
            ))}
          </select>

          {/* სორტირება / Sort */}
          <select
            value={sortBy}
            onChange={(e) => setSortBy(e.target.value)}
            className="filter-select"
          >
            <option value="newest">
              {lang === 'ka' ? 'უახლესი' : 'Newest'}
            </option>
            <option value="price-low">
              {lang === 'ka' ? 'ფასი: დაბლიდან მაღლა' : 'Price: Low to High'}
            </option>
            <option value="price-high">
              {lang === 'ka' ? 'ფასი: მაღლიდან დაბლა' : 'Price: High to Low'}
            </option>
            <option value="rating">
              {lang === 'ka' ? 'რეიტინგი' : 'Rating'}
            </option>
          </select>

          {/* ძიება / Search */}
          <input
            type="text"
            placeholder={lang === 'ka' ? 'ძიება...' : 'Search...'}
            value={searchQuery}
            onChange={(e) => setSearchQuery(e.target.value)}
            className="search-input"
          />
        </div>

        {/* შედეგების რაოდენობა / Results count */}
        <p className="results-count">
          {lang === 'ka' 
            ? `ნაპოვნია ${filteredProducts.length} პროდუქტი`
            : `Found ${filteredProducts.length} products`}
        </p>

        {/* პროდუქტების გრიდი / Products Grid */}
        <div className="products-grid">
          {filteredProducts.length > 0 ? (
            filteredProducts.map(product => (
              <ProductCard key={product.id} product={product} />
            ))
          ) : (
            <div className="no-results">
              <p>
                {lang === 'ka' 
                  ? 'პროდუქტი ვერ მოიძებნა' 
                  : 'No products found'}
              </p>
            </div>
          )}
        </div>
      </div>
    </>
  );
};

export default ProductsPage;
```
